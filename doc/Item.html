def get_standard_price
  Rails.cache.fetch([self, 'standard_price']) do
    standard_url = self.get_urls & self.group.get_standard_site.get_urls
    if standard_url.nil? || standard_url.empty?
      nil
    else
      standard_url.first.price
    end
  end
end
